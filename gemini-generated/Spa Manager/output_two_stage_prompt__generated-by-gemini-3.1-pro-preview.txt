    state SpaManager {
        region SaunaRegion {
            state SaunaOff
            state SaunaOn {
                region HeatingRegion {
                    state HeatingOn
                    state HeatingOff
                    [*] --> HeatingOn
                    
                    note right of HeatingOn
                        do: heat()
                    end note
                    
                    HeatingOn --> HeatingOff : tempUpdate [temp >= 90]
                    HeatingOff --> HeatingOn : tempUpdate [temp < 85]
                }
                region FanRegion {
                    state FanOff
                    state FanMonitoring
                    state FanOn
                    [*] --> FanOff
                    
                    note right of FanOn
                        do: runFan()
                    end note
                    
                    FanOff --> FanMonitoring : humidityUpdate [humidity > 40]
                    FanMonitoring --> FanOff : humidityUpdate [humidity <= 40]
                    FanMonitoring --> FanOn : after(3min)
                    FanOn --> FanOff : after(5min)
                }
                region WaterRegion {
                    state WaterReady
                    state WaterCooldown
                    [*] --> WaterReady
                    
                    WaterReady --> WaterCooldown : requestWater [humidity < 40 && !fanActive] / disperseWater()
                    WaterCooldown --> WaterReady : after(15min)
                }
            }
            [*] --> SaunaOff
            
            SaunaOff --> SaunaOn : turnOnSauna
            SaunaOn --> SaunaOff : turnOffSauna
        }
        
        region JacuzziRegion {
            state JacuzziOff
            state JacuzziOn {
                state JacuzziActive {
                    region PressureRegion {
                        state Level1
                        state Level2
                        state Level3
                        [*] --> Level1
                        
                        Level1 --> Level2 : increase
                        Level2 --> Level3 : increase
                        Level3 --> Level2 : decrease
                        Level2 --> Level1 : decrease
                    }
                    region PatternRegion {
                        state PatternSelected
                        [*] --> PatternSelected
                        
                        PatternSelected --> PatternSelected : selectPattern(p) [p >= 1 && p <= 25] / setPattern(p)
                    }
                    state H
                }
                
                note right of JacuzziActive
                    do: ejectWater()
                end note
                
                state JacuzziPaused
                [*] --> JacuzziActive
                
                JacuzziActive --> JacuzziPaused : pause
                JacuzziPaused --> H : after(2min)
            }
            [*] --> JacuzziOff
            
            JacuzziOff --> JacuzziOn : turnOnJacuzzi
            JacuzziOn --> JacuzziOff : turnOffJacuzzi
        }
    }
    [*] --> SpaManager
