stateDiagram-v2
    state Off
    state On
    
    [*] --> Off
    Off --> On : turnOn
    On --> Off : turnOff
    
    state On {
        state Setup {
            region CourseSelection {
                state SelectingCourse
                [*] --> SelectingCourse
                SelectingCourse --> SelectingCourse : menu / nextCourse()
            }
            region CrustSelection {
                state Medium
                state Dark
                state Light
                
                [*] --> Medium
                Medium --> Dark : crust
                Dark --> Light : crust
                Light --> Medium : crust
            }
            region TimerSetup {
                state SettingTimer
                [*] --> SettingTimer
                SettingTimer --> SettingTimer : plus [timer < 18h] / increaseTimer(10min)
                SettingTimer --> SettingTimer : minus [timer > 0] / decreaseTimer(10min)
            }
        }
        
        note right of Setup
            entry / setCourse(1)
            entry / setTimer(0)
        end note
        
        state Countdown
        
        state BakingProcess {
            state Kneading
            state Rising
            state Baking
            state Warming
            
            [*] --> Kneading
            Kneading --> Rising : after(kneadingTime)
            Rising --> Baking : after(risingTime)
            Baking --> Warming : after(bakingTime)
            
            note right of Kneading
                entry / startKneading()
                exit / stopKneading()
            end note
            
            note right of Baking
                do: heat()
            end note
            
            note right of Warming
                do: warm()
            end note
        }
        
        [*] --> Setup
        
        Setup --> Countdown : start [timer > 0]
        Setup --> BakingProcess : start [timer == 0]
        
        Countdown --> BakingProcess : timerFinished
        Countdown --> Setup : stop
        
        BakingProcess --> Setup : stop
        Warming --> Setup : after(60min)
    }
