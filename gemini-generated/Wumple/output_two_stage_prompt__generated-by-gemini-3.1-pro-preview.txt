    state Watch {
        region Main {
            state NormalOperation {
                state Timekeeping {
                    state ShowTime
                    state SetTime {
                        state SetSeconds
                        state SetMinutes
                        state SetHours

                        [*] --> SetSeconds
                        SetSeconds --> SetMinutes : C
                        SetMinutes --> SetHours : C
                        SetHours --> SetSeconds : C
                        
                        SetSeconds --> SetSeconds : B / incrementSeconds()
                        SetMinutes --> SetMinutes : B / incrementMinutes()
                        SetHours --> SetHours : B / incrementHours()
                    }

                    [*] --> ShowTime
                    ShowTime --> SetTime : C
                    SetTime --> ShowTime : A / saveTime()
                    ShowTime --> ShowTime : B / toggleAlarm()
                    ShowTime --> ShowTime : B_held_2s / toggleFlashAlert()
                }

                state Alarm {
                    state ShowAlarm
                    state SetAlarm {
                        state SetAlarmMinutes
                        state SetAlarmHours

                        [*] --> SetAlarmMinutes
                        SetAlarmMinutes --> SetAlarmHours : C
                        SetAlarmHours --> SetAlarmMinutes : C
                        
                        SetAlarmMinutes --> SetAlarmMinutes : B / incrementMinutes()
                        SetAlarmHours --> SetAlarmHours : B / incrementHours()
                    }

                    [*] --> ShowAlarm
                    ShowAlarm --> SetAlarm : C
                    SetAlarm --> Timekeeping : A / saveAlarm()
                }

                state Countdown {
                    state ShowCountdown
                    state SetCountdown {
                        state SetCDSeconds
                        state SetCDMinutes
                        state SetCDHours

                        [*] --> SetCDSeconds
                        SetCDSeconds --> SetCDMinutes : C
                        SetCDMinutes --> SetCDHours : C
                        SetCDHours --> SetCDSeconds : C
                        
                        SetCDSeconds --> SetCDSeconds : B / incrementSeconds()
                        SetCDMinutes --> SetCDMinutes : B / incrementMinutes()
                        SetCDHours --> SetCDHours : B / incrementHours()
                    }
                    state CountdownRunning

                    [*] --> ShowCountdown
                    ShowCountdown --> SetCountdown : C
                    SetCountdown --> ShowCountdown : A / saveCountdown()
                    ShowCountdown --> CountdownRunning : B [countdownTimeSet] / startCountdown()
                    CountdownRunning --> ShowCountdown : B / stopCountdown()
                }

                state H

                [*] --> Timekeeping
                
                ShowTime --> Alarm : A
                ShowAlarm --> Countdown : A
                ShowCountdown --> Timekeeping : A
                CountdownRunning --> Timekeeping : A
            }

            state AlarmRinging
            state CountdownRinging

            [*] --> NormalOperation
            
            NormalOperation --> AlarmRinging : alarmTimeReached
            AlarmRinging --> H : anyButton
            AlarmRinging --> H : after_5s

            NormalOperation --> CountdownRinging : countdownReachedZero
            CountdownRinging --> Countdown : anyButton
            CountdownRinging --> Countdown : after_5s
        }

        region Backlight {
            state BacklightOff
            state BacklightOn
            state BacklightFlashing

            [*] --> BacklightOff
            
            BacklightOff --> BacklightOn : D
            BacklightOn --> BacklightOff : D
            BacklightOn --> BacklightOff : after_10s

            BacklightOff --> BacklightFlashing : alarmTimeReached [flashAlertOn]
            BacklightOn --> BacklightFlashing : alarmTimeReached [flashAlertOn]
            BacklightOff --> BacklightFlashing : countdownReachedZero [flashAlertOn]
            BacklightOn --> BacklightFlashing : countdownReachedZero [flashAlertOn]

            BacklightFlashing --> BacklightOff : anyButton
            BacklightFlashing --> BacklightOff : after_5s
        }
    }

    note right of SetSeconds
        do: flash()
    end note
    
    note right of SetMinutes
        do: flash()
    end note
    
    note right of SetHours
        do: flash()
    end note
    
    note right of SetAlarmMinutes
        do: flash()
    end note
    
    note right of SetAlarmHours
        do: flash()
    end note
    
    note right of SetCDSeconds
        do: flash()
    end note
    
    note right of SetCDMinutes
        do: flash()
    end note
    
    note right of SetCDHours
        do: flash()
    end note
    
    note right of AlarmRinging
        do: beepAndFlashAlarmTime()
    end note
    
    note right of CountdownRinging
        do: beepAndFlashZeros()
    end note

    [*] --> Watch
