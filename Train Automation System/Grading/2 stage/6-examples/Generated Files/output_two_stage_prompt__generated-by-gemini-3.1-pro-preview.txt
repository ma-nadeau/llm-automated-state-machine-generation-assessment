stateDiagram-v2
    state Scheduled
    state Active {
        region ControlRegion {
            state Control {
                state AtStation
                note right of AtStation
                    entry / openDoors()
                end note
                state Moving
                state StoppingAtLight
                state WaitingAtLight
                state StoppingAtStation
                state H
                
                [*] --> AtStation
                AtStation --> Moving : after(stopDuration) [!finalStation] / closeDoors(), setSpeed(currentSegment.speed)
                Moving --> Moving : newSegment [regular || (trafficLight && green)] / setSpeed(currentSegment.speed)
                Moving --> StoppingAtLight : newSegment [trafficLight && red] / stopAL(currentSegment.location)
                Moving --> StoppingAtStation : newSegment [station] / stopAL(currentSegment.location)
                StoppingAtLight --> WaitingAtLight : stopped
                WaitingAtLight --> Moving : green / setSpeed(currentSegment.speed)
                StoppingAtStation --> AtStation : stopped
            }
            state Emergency
            
            [*] --> Control
            Control --> Emergency : obstructed / eStop()
            Emergency --> H : cleared
        }
        region SDSRegion {
            state SDS_Running
            [*] --> SDS_Running
            SDS_Running --> SDS_Running : after(1s) [newSegmentDetected] / setCS(segment)
        }
        region MDSRegion {
            state MDS_Running
            [*] --> MDS_Running
            MDS_Running --> MDS_Running : after(1s) / checkMovement()
        }
        region ODSRegion {
            state ODS_Running
            [*] --> ODS_Running
            ODS_Running --> ODS_Running : after(1s) / checkObstruction()
        }
    }
    state Off
    
    [*] --> Scheduled
    Scheduled --> Active : atStartOfStopDuration / pOn(), setCS(firstSegment)
    AtStation --> Off : after(stopDuration) [finalStation] / pOff()
